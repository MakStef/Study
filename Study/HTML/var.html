                import os
                from sys import exit
                from time import sleep
                
                def check():
                    global directory, path
                    os.chdir(path)
                    os.system("git status")
                
                def add():
                    global directory
                    os.system(f"git add {directory}")
                
                def commit():
                    msg=input("Введи повідомлення для коміту:\n")
                    os.system(f'git commit -m "{msg}"')
                
                def push():
                    global remote, branch
                    os.system(f"git push {remote} {branch}")

У ній ключовими для адаптації до різних умов є чотири змінні: path, directory, remote і branch, котрі задають відповідно шлях(path) до репозиторію, папкуdirectory), в котрій потрібно здійснити комміт, пульт(remote) керування і гілку(branch), котру штовхається.
